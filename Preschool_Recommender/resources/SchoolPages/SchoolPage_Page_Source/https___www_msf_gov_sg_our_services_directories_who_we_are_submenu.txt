    function calDistance(lat1, lon1, lat2, lon2, unit) {
        var radlat1 = Math.PI * lat1 / 180
        var radlat2 = Math.PI * lat2 / 180
        var theta = lon1 - lon2
        var radtheta = Math.PI * theta / 180
        var dist = Math.sin(radlat1) * Math.sin(radlat2) + Math.cos(radlat1) * Math.cos(radlat2) * Math.cos(radtheta);
        if (dist > 1) {
            dist = 1;
        }
        dist = Math.acos(dist)
        dist = dist * 180 / Math.PI
        dist = dist * 60 * 1.1515
        if (unit == "K") { dist = dist * 1.609344 }
        if (unit == "N") { dist = dist * 0.8684 }
        return dist
    }

    // #region render html marker
    function renderMarker(markerData, officesListEle, typeService) {
        let officeItem = officeInfoTemplate
            .replace(/{OFFICE_ID}/g, markerData?.options?.postalCode)
            .replace('{TITLE}', markerData?.options?.title)
            .replace('{SERVICE_TYPE}', typeService);

        const addr = markerData?.options?.address?.replace(/^\s+|\s+$/gm,'');
        if (!!addr) {
            officeItem = officeItem.replace('{DISPLAY_ADDRESS}', 'display-block');
            officeItem = officeItem.replace('{ADDRESS}', addr);
        } else {
            officeItem = officeItem.replace('{DISPLAY_ADDRESS}', 'display-none');
        }

        const phoneNum = markerData?.options?.phone?.replace(/^\s+|\s+$/gm,'');
        if (!!phoneNum) {
            officeItem = officeItem.replace('{DISPLAY_PHONE}', 'display-block');
            officeItem = officeItem.replace('{PHONE}', phoneNum);
        } else {
            officeItem = officeItem.replace('{DISPLAY_PHONE}', 'display-none');
        }

        const email = markerData?.options?.email?.replace(/^\s+|\s+$/gm,'');
        if (!!email) {
            officeItem = officeItem.replace('{DISPLAY_EMAIL}', 'display-block');
            officeItem = officeItem.replace(/{EMAIL}/g, email);
        } else {
            officeItem = officeItem.replace('{DISPLAY_EMAIL}', 'display-none');
        }

        const link = markerData?.options?.hyperLink?.replace(/^\s+|\s+$/gm,'');
        if (!!link) {
            officeItem = officeItem.replace('{DISPLAY_WEBSITE}', 'display-block');
            officeItem = officeItem.replace(/{WEBSITE}/g, link);
        } else {
            officeItem = officeItem.replace('{DISPLAY_WEBSITE}', 'display-none');
        }

        const fax = markerData?.options?.fax?.replace(/^\s+|\s+$/gm,'');
        if (!!fax) {
            officeItem = officeItem.replace('{DISPLAY_FAX}', 'display-block');
            officeItem = officeItem.replace('{FAX_NUMBER}', fax);
        } else {
            officeItem = officeItem.replace('{DISPLAY_FAX}', 'display-none');
        }

        const hours = markerData?.options?.operatingHours?.replace(/^\s+|\s+$/gm,'');
        if (!!hours) {
            officeItem = officeItem.replace('{DISPLAY_OPERATING_HOURS}', 'display-block');
            officeItem = officeItem.replace('{OPERATING_HOURS}', hours);
        } else {
            officeItem = officeItem.replace('{DISPLAY_OPERATING_HOURS}', 'display-none');
        }

        const scfaAdmin = markerData?.options?.scfaAdmin;
        if (scfaAdmin === null || scfaAdmin === undefined) {
            officeItem = officeItem.replace('{DISPLAY_SCFAADMIN}', 'display-none');
        } else {
            officeItem = officeItem.replace('{DISPLAY_SCFAADMIN}', 'display-block');
            officeItem = officeItem.replace('{SCFAADMIN}', scfaAdmin);
        }

        const fee = markerData?.options?.monthlyFees?.replace(/^\s+|\s+$/gm,'');
        if (!!fee) {
            const feeNum = Number(fee);
            const feeString = !feeNum ? fee : `$${feeNum.toFixed(2)}`;
            officeItem = officeItem.replace('{DISPLAY_MONTHLYFEES}', 'display-block');
            officeItem = officeItem.replace('{MONTHLYFEES}', feeString);
        } else {
            officeItem = officeItem.replace('{DISPLAY_MONTHLYFEES}', 'display-none');
        }

        const auditDate = markerData?.options?.dateOfAudit?.replace(/^\s+|\s+$/gm,'');
        const auditStatus = markerData?.options?.auditStatus?.replace(/^\s+|\s+$/gm,'');
        if (!!auditDate || !!auditStatus) {
            officeItem = officeItem.replace('{DISPLAY_AUDITSTATUSWRAPPER}', 'display-block');
        } else {
            officeItem = officeItem.replace('{DISPLAY_AUDITSTATUSWRAPPER}', 'display-none');
        }
        if (!!auditDate) {
            officeItem = officeItem.replace('{DISPLAY_DATEOFSTATUS}', 'display-block');
            officeItem = officeItem.replace('{DATEOFSTATUS}', auditDate);
        } else {
            officeItem = officeItem.replace('{DISPLAY_DATEOFSTATUS}', 'display-none');
        }
        if (!!auditStatus) {
            officeItem = officeItem.replace('{DISPLAY_AUDITSTATUS}', 'display-block');
            officeItem = officeItem.replace('{AUDITSTATUS}', auditStatus);
        } else {
            officeItem = officeItem.replace('{DISPLAY_AUDITSTATUS}', 'display-none');
        }

        const latestQualityAuditHistory = markerData?.options?.latestQualityAuditHistory?.replace(/^\s+|\s+$/gm,'');
        if (!!latestQualityAuditHistory) {
            officeItem = officeItem.replace('{DISPLAY_AUDITHISTORYWRAPPER}', 'display-block');
            officeItem = officeItem.replace('{DISPLAY_AUDITHISTORY}', 'display-block');
            officeItem = officeItem.replace('{AUDITHISTORY}', latestQualityAuditHistory);
        } else {
            officeItem = officeItem.replace('{DISPLAY_AUDITHISTORYWRAPPER}', 'display-none');
            officeItem = officeItem.replace('{DISPLAY_AUDITHISTORY}', 'display-none');
        }

        const footnote = markerData?.options?.footnote?.replace(/^\s+|\s+$/gm,'');
        if (!!footnote) {
            officeItem = officeItem.replace('{DISPLAY_FOOTNOTEWRAPPER}', 'display-block');
            officeItem = officeItem.replace('{DISPLAY_FOOTNOTE}', 'display-block');
            officeItem = officeItem.replace('{FOOTNOTE}', footnote);
        } else {
            officeItem = officeItem.replace('{DISPLAY_FOOTNOTEWRAPPER}', 'display-none');
            officeItem = officeItem.replace('{DISPLAY_FOOTNOTE}', 'display-none');
        }

        const des = markerData?.options?.description?.replace(/^\s+|\s+$/gm,'');
        if (!!des) {
            if (typeService === "STRENGTHENING_SERVICE" || typeService === "STUDENT_CARE_SERVICE") {
                officeItem = officeItem.replace('{DISPLAY_OTHER_INFO}', 'display-block');
            } else {
                officeItem = officeItem.replace('{DISPLAY_OTHER_INFO}', 'display-none');
            }
            officeItem = officeItem.replace('{OTHER_INFO}', des);
        } else {
            officeItem = officeItem.replace('{DISPLAY_OTHER_INFO}', 'display-none');
        }
        officesListEle.innerHTML += officeItem;
    }

    // #region button show/hide detail
    function checkShowButtonMoreDetail(typeService) {
        const listEle = $(`#officesList_${typeService} .content-wrapper`);
        if (!listEle.length) return;
        for (var i = 0; i < listEle.length; i++) {
            const id = listEle[i]?.id;
            const uniqueKey = id.replace("content_", "");
            const heightShow = $(`#${id}`).innerHeight() + 3;
            const heightOriginal = Math.round($(`#${id} .main-content`).innerHeight()) + 39;
            const isDisplay = heightShow < heightOriginal;

            $(`#uniqueOffice-${uniqueKey} .read-more`).removeClass(isDisplay ? "display-none" : "display-block");
            $(`#uniqueOffice-${uniqueKey} .read-more`).addClass(isDisplay ? "display-block" : "display-none");
        }
    }
    function toggleMoreLessAction(id, typeService, isForceShow = false, page, ) {
        if (page > 0) {
            if (currentPage !== page) changePage(page);
        }
        setTimeout(() => {
            const defaultHeight = typeService === "STRENGTHENING_SERVICE" ? 187 : 185 + 3;
            const heightShow = $(`#content_${id}`).innerHeight();

            const heightOriginal = Math.round($(`#content_${id} .main-content`).innerHeight()) + 39;
            const isShow = isForceShow ? true : heightShow < heightOriginal;

            document.getElementById(`content_${id}`).style.height = isShow ? `${heightOriginal}px` : `${defaultHeight}px`;
            const textEle = $(`#uniqueOffice-${id} .read-more .text`);
            const arrowEle = $(`#uniqueOffice-${id} .read-more .arrow`);
            if (isShow) {
                if (!textEle.hasClass("show")) textEle.addClass("show");
                if (!arrowEle.hasClass("show")) arrowEle.addClass("show");
            } else {
                textEle.removeClass("show");
                arrowEle.removeClass("show");
            }
        }, 300);
    }

        const map = getMapByTypeService(typeService);
        const markerList = getNearYouMarketListByTypeService(typeService);
        if (!markerList.length) return;

        markerList.forEach((item) => {
            renderMarker(item, officesListEle, typeService);
            addMarker(item, {
                type: item?.options?.type,
                title: item?.options?.title,
                address: item?.options?.address,
                postalCode: item?.options?.postalCode,
                page: item?.options?.page || -1,
                serviceType: typeService,
            }, map);
        });

        officeServingYou.style.display = "block";
    }

        const map = getMapByTypeService(typeService);
        const markerList = getNearYouMarketListByTypeService(typeService);

        const element = setOfficeNearYouServiceElement(typeService);
        const officesListEle = element.officesListEle;
        const officeServingYou = element.officeServingYou;

        markerList.forEach((item) => {
            map.removeLayer(item);
        });

        officeServingYou.style.display = "none";
    }

    function filterAction(typeService, isChangeDistance = false) {
        setTimeout(() => {
            isFilterByDistance = isChangeDistance;
            toggleTriagle(typeService, false);
            toggleMsgNoData(typeService, false);

            const isMobile = checkIsMobile();

            let filterType;
            if (typeService !== 'STUDENT_CARE_SERVICE') {
                filterType = "POSTAL";
            } else {
                filterType = isMobile
                    ? $(`#filterTypePostalMobile_${typeService} li.active`).attr('data-value')
                    : $(`#filterTypePostal_${typeService} li.active`).attr('data-value');

                if (!isChangeDistance) {
                    changeDistanceHtmlAction(4, 'STUDENT_CARE_SERVICE')
                }
                toggleDistance(typeService, false);
            }

            setOfficeServiceElement(typeService);
            let markerList = getMarketListByTypeService(typeService);
            let nearMarkerList = getNearYouMarketListByTypeService(typeService);


            const searchValue = isMobile
                ? $(`#txtSearchPostalMobile_${typeService}`).val()?.toLowerCase().replace(/^\s+|\s+$/gm,'')
                : $(`#txtSearchPostal_${typeService}`).val()?.toLowerCase().replace(/^\s+|\s+$/gm,'');
            const map = getMapByTypeService(typeService);
            if (markerList.length > 0) {
                markerList.forEach((item) => {
                    map.removeLayer(item);
                });
            }
            if (typeService !== "STUDENT_CARE_SERVICE" && nearMarkerList.length > 0) {
                nearMarkerList.forEach((item) => {
                    map.removeLayer(item);
                });
            }
            clearMarket(typeService);

            if (!searchValue) return;

            const distance = typeService != "STUDENT_CARE_SERVICE"
                ? 2
                : $(`#distance_${typeService} .swiper-slide.active`).attr('data-distance');

            switch (filterType) {
                case 'POSTAL':
                    callApiGetMarker(typeService, searchValue, distance);
                    break;
                case 'OFFICE_NAME':
                    callApiFilterStudent(searchValue.toLowerCase().replace(/^\s+|\s+$/gm, ''), '', filterType);
                    break;
                case 'OFFICE_ADDRESS':
                    callApiFilterStudent('', searchValue.toLowerCase().replace(/^\s+|\s+$/gm, ''), filterType);
                    break;
                default:
                    break;
            }
        }, 300)
    }

    function callApiFilterStudent(name, address, filterType) {
        if (!name && !address) return;

        let data = {
            "name": name,
            "address": address,
        };
        $.ajax({
            url: "/customapi/Dfcs/FilterDfcsStudent",
            method: "post",
            data: data
        }).done(function (resp) {

            if (!resp?.length) {
                toggleMsgNoData('STUDENT_CARE_SERVICE', true);
                return;
            }

            let postalList = resp.map(x => x.ADDRESSPOSTALCODE);
            callSSCOnemapApi(postalList, postalList, [], resp, filterType);
        });
    }

    function callSSCOnemapApi(originalPostalList, postalList, markerList, resp, filterType) {
        const token = 'eyJ0eXAiOiJKV1QiLCJhbGciOiJIUzI1NiJ9.eyJzdWIiOiJlNTg1NTM3MzE2MmM5ZmQ0YzI0NDRlNjliMmJhOWE5YiIsImlzcyI6Imh0dHA6Ly9pbnRlcm5hbC1hbGItb20tcHJkZXppdC1pdC0xMjIzNjk4OTkyLmFwLXNvdXRoZWFzdC0xLmVsYi5hbWF6b25hd3MuY29tL2FwaS92Mi91c2VyL3Nlc3Npb24iLCJpYXQiOjE3MTE4Mzg4NzksImV4cCI6MTcxMjA5ODA3OSwibmJmIjoxNzExODM4ODc5LCJqdGkiOiJzbnZsaWZiQTcyT212dkJmIiwidXNlcl9pZCI6ODU4LCJmb3JldmVyIjpmYWxzZX0.dM1GUXBTEZWX_rgUm0xDNqX0lvv16G6WQNDEmi2fTPE';
        if (!token) {
            alert("Cannot get onemap api token!");
            return;
        }

        return new Promise((resolve, reject) => {
            let markerDataTemp = markerList;
            const postal = postalList[0];
            $.ajaxSetup({
                headers: {
                    'Authorization': `Bearer ${token}`
                }
            });
            $.get(`https://www.onemap.gov.sg/api/msf/getStudentCare?postal=${postal}&distance=10`, function (data) {
                postalList = postalList.filter(x => x !== postal);
                if (!data?.SrchResults?.length) return;

                for (var j = 0; j < data?.SrchResults?.length; j++) {
                    const item = data?.SrchResults[j];

                    const isExistCurrentPostal = postal === item.ADDRESSPOSTALCODE || postal === `0${item.ADDRESSPOSTALCODE}`;
                    if (isExistCurrentPostal) {
                        markerDataTemp.push(item);
                        continue;
                    }

                    const existPostal = postalList.find(x => x === item.ADDRESSPOSTALCODE || x === `0${item.ADDRESSPOSTALCODE}`);
                    if (!existPostal) continue;

                    postalList = postalList.filter(x => x !== existPostal);
                    markerDataTemp.push(item);
                }
                if (postalList.length === 0) {
                    handleStudentDataAfterCallApi(resp, studentCareMap, markerDataTemp, filterType, '');
                    resolve(markerDataTemp);
                } else {
                    callSSCOnemapApi(originalPostalList, postalList, markerDataTemp, resp, filterType);
                }
            }, "json")
            .fail(function (err) {
                postalList = postalList.filter(x => x !== postal);
                if (postalList.length === 0) {
                    resolve(markerDataTemp)
                } else {
                    callSSCOnemapApi(originalPostalList, postalList, markerDataTemp, resp, filterType);
                }
            });
        });
    }

    function handleDataAfterCallApi(localData, map, typeService, isNearYouService, onemapData) {
        if (!localData?.length) {
            if (isNearYouService) {
                const serviceYouMarkerList = getMarketListByTypeService(typeService);
                if (!serviceYouMarkerList?.length) toggleMsgNoData(typeService, true);
            }
            return;
        }

        const markerList = isNearYouService
            ? getNearYouMarketListByTypeService(typeService) : getMarketListByTypeService(typeService);

        const element = isNearYouService
            ? setOfficeNearYouServiceElement(typeService) : setOfficeServiceElement(typeService);
        const officesListEle = element.officesListEle;
        const officeServingYou = element.officeServingYou;

        let firstLat = '';
        let firstLng = '';
        let isSetLatLong = false;
        const isShowNearYou = isNearYouService ? $(`#flexSwitchCheckDefault_${typeService}`).is(":checked") : false;
        for (var i = 0; i < localData.length; i++) {
            const result = localData[i];

            const oneMapItem = onemapData.find(x => x.ADDRESSPOSTALCODE === result.ADDRESSPOSTALCODE || `0${x.ADDRESSPOSTALCODE}` === result.ADDRESSPOSTALCODE);
            result.LatLng = oneMapItem?.LatLng;
            if (!result.LatLng) continue;

            const latLngList = result.LatLng.split(',');
            const lat = parseFloat(latLngList[0]);
            const lng = parseFloat(latLngList[1]);
            if (!!lat && !!lng && !isSetLatLong) {
                firstLat = lat;
                firstLng = lng;
                isSetLatLong = true;
            }

            const markerData = {
                title: result.NAME || "",
                type: getMarkerCentreType(typeService),
                address: result.ADDRESS || "",
                postalCode: result.ADDRESSPOSTALCODE || "",
                description: result.DESCRIPTION || "",
                email: result.EMAILADDRESS || "",
                phone: result.NUMBER || "",
                fax: result?.FAXNUMBER || "",
                hyperLink: result.WEBSITEADDRESS || "",
                operatingHours: result.OPERATINGHOURS || "",
                icon: isNearYouService ? nearYoumarkerIcon : markerIcon,
                alt: 'marker'
            };

            if (!isNearYouService || (isNearYouService && isShowNearYou)) {
                renderMarker(marker, officesListEle, typeService);
                addMarker(marker, {
                    type: marker?.options?.type,
                    title: marker?.options?.title,
                    address: marker?.options?.address,
                    postalCode: marker?.options?.postalCode,
                    page: marker?.options?.page || -1,
                    serviceType: typeService,
                }, map);
            }
            markerList.push(marker);
        }

        if (!isNearYouService && !!firstLat && !!firstLng) {
            map.flyTo([firstLat, firstLng]);
        }
        if (!isNearYouService || (isNearYouService && isShowNearYou)) {
            officeServingYou.style.display = "block";
            checkShowButtonMoreDetail(typeService);
        }

        // show triangle
        toggleTriagle(typeService, true);

    // #region PAGNATION
    function calcPageForItem(index) {
        const pageTmp = Math.floor(index / itemPerPage);
        return pageTmp + 1;
    }
    function previousAction() {
        changePage(currentPage - 1);
    }
    function nextAction() {
        changePage(currentPage + 1);
    }
    function changePage(page) {
        currentPage = page ? page : 1;
        const startIndex = (currentPage - 1) * itemPerPage;
        const endIndex = startIndex + itemPerPage;

        const currentTypeService = $("#scroller .nav-link.active").data("key");
        const currentMarkerList = getMarketListByTypeService(currentTypeService);

        const element = setOfficeServiceElement(currentTypeService);
        const officesListEle = element.officesListEle;
        const officeServingYou = element.officeServingYou;

        for (var i = startIndex; i < endIndex; i++) {
            if (i === currentMarkerList.length) break;

            renderMarker(currentMarkerList[i], officesListEle, currentTypeService);
        }
        officeServingYou.style.display = "block";
        checkShowButtonMoreDetail(currentTypeService);
        generatePanation(currentMarkerList, currentTypeService);
    }

    function generatePanation(data, typeService) {
        const totalPage = Math.ceil(data.length / itemPerPage);

        const pageQuantityShow = 3;
        let pageNumberHtml = "";

        let startPage;
        if (currentPage === 1 || totalPage < pageQuantityShow + 1) {
            startPage = 1;
        } else if (currentPage === totalPage) {
            startPage = currentPage - (pageQuantityShow - 1);
        } else if (currentPage - 1 >= 1) {
            startPage = currentPage - 1;
        }

        if (startPage > 2) {
            pageNumberHtml += templatePagnation
                .replace('PARAM_ACTIVE_PAGE', '')
                .replace('PARAM_PAGE_CHANGE', 1)
                .replace('PARAM_PAGE_NUMBER', 1);
        }
        if (startPage > 1) {
            pageNumberHtml += threeDotText;
        }

        const endPage = startPage + (pageQuantityShow - 1) >= totalPage ? totalPage : startPage + (pageQuantityShow - 1);
        for (var i = startPage; i <= endPage; i++) {
            pageNumberHtml += templatePagnation
                .replace('PARAM_ACTIVE_PAGE', i === currentPage ? 'active' : '')
                .replace('PARAM_PAGE_CHANGE', i)
                .replace('PARAM_PAGE_NUMBER', i);
        }
        if (totalPage > endPage + 1) {
            pageNumberHtml += threeDotText;
        }
        if (totalPage > endPage) {
            pageNumberHtml += templatePagnation
                .replace('PARAM_ACTIVE_PAGE', '')
                .replace('PARAM_PAGE_CHANGE', totalPage)
                .replace('PARAM_PAGE_NUMBER', totalPage);
        }

        let previousHtml = templatePrevious.replace('PARAM_PREVIOUS_DISABLE', currentPage === 1 ? 'disabled' : '');
        let nexthtml = templateNext.replace('PARAM_NEXT_DISABLE', currentPage === totalPage ? 'disabled' : '');

        document.getElementById(`pagnationUl_${typeService}`).innerHTML = "";
        document.getElementById(`pagnationUl_${typeService}`).innerHTML = previousHtml + pageNumberHtml + nexthtml;
    }
