        function cleanText(val) {
            if (val != "" && val != null) {
                return val.replace(/[^a-zA-Z]+/g, "").toLowerCase();
            }

        function assignTag(val, init) {
            if (val === "object") {
                if (val.type != "" && val.type == null) {
                    val = val.type;
                }
                else if(val.topic != "" && val.topic == null)
                {
                    val = val.topic
                }
            }

            var t_find = filters_topic.find(x => cleanText(x.value) == cleanText(JSON.stringify(val)));
            if (t_find) {
                $('#topicSelect').val(t_find.value);
                if (init == "false") {
                    $('#topicSelect').selectmenu('refresh');
                }
            }
            var p_find = filters_type.find(x => cleanText(x.value) == cleanText(JSON.stringify(val)));
            if (p_find) {

    function changePage(page) {
            currentPage = page ? page : 1;
            const startIndex = (currentPage - 1) * itemPerPage;
            const endIndex = startIndex + itemPerPage;
            document.getElementById('media-room-result').innerHTML = "";
            for (var i = startIndex; i < endIndex; i++) {
                if (i === currentData.length) break;
                currentData[i].mediaTopic.replace("&amp;", "&");
                currentData[i].mediaCategory.replace("&amp;", "&");
                TopicData = currentData[i].mediaTopic.split(";");
                CategoryData = currentData[i].mediaCategory.split(";")
                totalCategorys = CategoryData.length - 1;
                totaltopics = TopicData.length - 1;
                if (TopicData.length == 2) {
                    TopicList = '<a href="#" class="card-type-link" onclick="filterMediaRoomTopics(\'' + TopicData[0] + '\' , \'' + "false" + '\')" aria-label="topic">' + TopicData[0] + '</a>';
                } else
                {
                    TopicData.forEach(GetTopicList);
                }
                if (CategoryData.length == 2) {
                    CategoryList = '<a href="#" class="card-type-link" onclick="filterMediaRoomCategory(\'' + CategoryData[0] + '\', \'' + "false" + '\')" aria-label="category">' + CategoryData[0] + '</a>';
                }
                else
                {
                    CategoryData.forEach(GetCategoryList);
                }
                document.getElementById('media-room-result').innerHTML += template
                    .replace('CARD_TITLE', currentData[i].title)
                    .replace('PageUrl', currentData[i].urlLink + "/" + currentData[i].urlName)
                    .replace('PUB_DATE', currentData[i].publishedDateStr)
                    .replace('SYNOPSIS', currentData[i].synopsis)
                    .replace('TOPICS', TopicList)
                    .replace('TYPE', CategoryList)
                    .replaceAll('TOPIC_ID', 'Topic-' + i)
                    .replace('Detail_ID', 'Detail-' + i) ;
                if (TopicData.length == 2)
                {
                    document.getElementById("Detail-" + i).style.display = "none";
                }
                CategoryList = "";
                TopicList = "";
        }
        generatePanation();
        }

        function generatePanation() {
            const totalPage = Math.ceil(currentData.length / itemPerPage);

            const pageQuantityShow = 3;
            let pageNumberHtml = "";

            let startPage;
            if (currentPage === 1 || totalPage < pageQuantityShow + 1) {
                startPage = 1;
            } else if (currentPage === totalPage) {
                startPage = currentPage - (pageQuantityShow - 1);
            }  else if (currentPage - 1 >= 1) {
                startPage = currentPage - 1;
            }

            if (startPage > 2) {
                pageNumberHtml += templatePagnation
                    .replace('PARAM_ACTIVE_PAGE', '')
                    .replace('PARAM_PAGE_CHANGE', 1)
                    .replace('PARAM_PAGE_NUMBER', 1);
            }
            if (startPage > 1) {
                pageNumberHtml += threeDotText;
            }

            const endPage = startPage + (pageQuantityShow - 1) >= totalPage ? totalPage : startPage + (pageQuantityShow - 1);
            for (var i = startPage; i <= endPage; i++) {
                pageNumberHtml += templatePagnation
                    .replace('PARAM_ACTIVE_PAGE', i === currentPage ? 'active' : '')
                    .replace('PARAM_PAGE_CHANGE', i)
                    .replace('PARAM_PAGE_NUMBER', i);
            }
            if (totalPage > endPage + 1) {
                pageNumberHtml += threeDotText;
            }
            if (totalPage > endPage) {
                pageNumberHtml += templatePagnation
                    .replace('PARAM_ACTIVE_PAGE', '')
                    .replace('PARAM_PAGE_CHANGE', totalPage)
                    .replace('PARAM_PAGE_NUMBER', totalPage);
            }

            let previousHtml = templatePrevious.replace('PARAM_PREVIOUS_DISABLE', currentPage === 1 ? 'disabled' : '');
            let nexthtml = templateNext.replace('PARAM_NEXT_DISABLE', currentPage === totalPage ? 'disabled' : '');

            document.getElementById('pagnationUl').innerHTML = "";
            document.getElementById('pagnationUl').innerHTML = previousHtml + pageNumberHtml + nexthtml;
        }
