                    .replace('PARAM_PAGE_CHANGE', 1)
                    .replace('PARAM_PAGE_NUMBER', 1);
            }
            if (startPage > 1) {
                pageNumberHtml += threeDotText;
            }

            const endPage = startPage + (pageQuantityShow - 1) >= totalPage ? totalPage : startPage + (pageQuantityShow - 1);
            for (var i = startPage; i <= endPage; i++) {
                pageNumberHtml += templatePagnation
                    .replace('PARAM_ACTIVE_PAGE', i === currentPage ? 'active' : '')
                    .replace('PARAM_PAGE_CHANGE', i)
                    .replace('PARAM_PAGE_NUMBER', i);
            }
            if (totalPage > endPage + 1) {
                pageNumberHtml += threeDotText;
            }
            if (totalPage > endPage) {
                pageNumberHtml += templatePagnation
                    .replace('PARAM_ACTIVE_PAGE', '')
                    .replace('PARAM_PAGE_CHANGE', totalPage)
                    .replace('PARAM_PAGE_NUMBER', totalPage);
            }

            let previousHtml = templatePrevious.replace('PARAM_PREVIOUS_DISABLE', currentPage === 1 ? 'disabled' : '');
            let nexthtml = templateNext.replace('PARAM_NEXT_DISABLE', currentPage === totalPage ? 'disabled' : '');

            document.getElementById('pagnationUl').innerHTML = "";
            document.getElementById('pagnationUl').innerHTML = previousHtml + pageNumberHtml + nexthtml;
        }

         * PAGNATION: END
         * */

            filterResearchReport(false);
