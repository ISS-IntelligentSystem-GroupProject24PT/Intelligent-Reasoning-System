function GetAlbum(page_index, jq)	//Gtype,1,代表学校,2,班级
{
	$("#School_album").html(GetLoading());
	$.get("aspcode/aspurl.html", { School_id: School_id, admin_id:admin_id,act: "School_album",currentpage:page_index+1, TopNum:Pnum,trNum:PPRow,ShowStrLen:ShowStrLen,Gtype:Gtype},
