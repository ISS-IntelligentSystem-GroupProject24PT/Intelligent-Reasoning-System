	 * @private

	 * @param {string} set1 Set of Emoji to test.

	function emojiSetsRenderIdentically( set1, set2 ) {

		context.clearRect( 0, 0, canvas.width, canvas.height );

		var rendered1 = canvas.toDataURL();
